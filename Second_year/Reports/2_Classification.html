def methodknn(traindata, testdata, k, numberofclasses):
    def distance(x,y):
        return math.sqrt((x[0] - y[0])**2 + (x[1] - y[1])**2)
    ans = []
    for testpoint in testdata:
        testdist = [[distance(testpoint,traindata[i]), int(traindata[i][2])] for i in range(len(traindata))]
        stat = [0 for i in range(numberofclasses)]
        for d in sorted(testdist)[0:k]:
            stat[d[1]] += 1
        ans.append(sorted(zip(stat, range(numberofclasses)),reverse=True)[0][1])
    return ans
